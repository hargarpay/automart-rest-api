    Introduction

    Welcome to the AutoMart API! You can use our API to access Automart API endpoints.

    You can view code examples in the dark area to the right.

    Authorization

    AutoMart uses API token to allow access to the API. To get the API token you need to create authorization.

    Create Authorization

    HTTP Request

    POST /api/v1/auth/signup

    Parameters

    Parameter Type Description
    first_name String User first name (Required)
    last_name String User last name (Required)
    phone String User phone (Required)
    address String User address (Optional)
    email String User email address (Required)
    password String User password (Required)
    address String User address
    street String User street
    city String User city
    state String User state
    country String User country
    phone String User phone

    Getting Authorization

    If you have registered and you need to be autheticated by using the email and password created to get the API token.

    HTTP Request

    POST /api/v1/auth/signin

    Parameters

    Parameter Type Description
    email String User email address (Required)
    password String User password (Required) 
    {
"success": true,
"status": "success",
"data": {
    "token": "meowmeowmeow",
    "id": 2,
    "first_name": "John",
    "last_name": "Smith",
    "email": "johnsmith@gmail.com",
    "phone": "2347011122233",
    "username": "johnsmith",
    "address": "test 2 street",
    "street": "test 2 street",
    "city": "test 2 city",
    "state": "test 2 state",
    "country": "Nigeria",
    "phone": "09088776644",
    "zip": "101-242",
  }
}
    Automart expects for the API token to be included in all API requests to the server in a header that looks like the following:
    x-access-token: Bearer meowmeowmeow

    You must replace meowmeowmeow with the return token after registration or authenticated (logging in).

    Car AD Image Upload

    Upload Car Image to image manager

    There is an image manager that handle all image upload in the application. Each seller only see image uploaded by them but the admin can see all image uploaded by all seller. It is from the image manager that the seller can assign image to the car ad. This gives the seller to assign the same image to many car ad without uploading the image multiple times. To implement the image at the frontend, the sample of the frontend is shown in the html and javascript section. All image are uploaded to cloudinary

    HTTP Request

    POST /api/v1/media

    Parameters

    Parameter Type Description
    start Number It should start from 0 Required
    filename String name of the file
    fileData String The file should be in base64 check the code sample to convert image file to base64
    uploading Boolean Image are upload in chunk (meaning little by little), so if it true it implies the base64 image is still uploading while if it false the base64 image has finish uploading
    extension Enum the extension of the file, the accepted value is jpg

    const slice_size = 5 * 1024; //5KB

var uploader = document.querySelector('.uploader');

if(uploader){
    uploader.addEventListener('change', function(e){
        var target = e.target;
        if(target.nodeName.toLowerCase() === 'input' && target.closest('.first-upload')){
            start_upload(target);
        }
    })
}

function start_upload( target ) {
    var wrapper = target.closest('.first-upload');
    var file = target.files[0];
    var spanMeter = wrapper.querySelector('span.meter');
    var progress = wrapper.querySelector('.progress');
    var reader = new FileReader();
    if(file){
        const now = Date.now();
        const filename = `filename-${now}-type`;
        const extension = file.name.split('.').pop().toLowerCase();
        const url = 'http://localhost:3000/api/v1/media';
        const config = {
            start: 0, filename, extension, url, file, spanMeter, progress,
            reader,
        }
        upload_file(config);
    }else{
        console.log('You have not uploaded any file');
    }
}

function upload_file(config) {
    const {
        start, filename, extension, url, file, spanMeter, progress, reader,
    } = config;
    var next_slice = start + slice_size + 1;
    var blob = file.slice( start, next_slice );
    var base64CodeSize = file.size;
    // var blob = file;
    reader.onloadend = function( event ) {
        if ( event.target.readyState !== FileReader.DONE ) {
            return;
        }
        var uploading = (start + slice_size) < base64CodeSize ? true : false; 
        console.log(event.target.result);
        const data = {
            fileData: event.target.result, filename, start,  extension, uploading
        }
        fetch(url, {
            method: 'POST',
            body: JSON.stringify(data),
            headers:{
                'Content-Type': 'application/json',
                Authorization: 'Bearer '+ token
            }
        })
        .then((resp) => {
            return resp.json();
        })
        .then((resp) => {
            console.log(resp);
            if(resp.success){
                if ( next_slice < base64CodeSize ) {
                    // Update upload progress
                    var size_done = start + slice_size;
                    var percent_done = Math.floor( ( size_done / base64CodeSize ) * 100 );
                    progress.value = percent_done;
                    spanMeter.textContent = percent_done + '%';
                    // More to upload, call function recursively
                    const innerConfig = {
                        start: next_slice, filename, extension, url, file,
                        spanMeter, progress, reader,
                    }
                    upload_file(innerConfig);
                } else {
                    progress.value = 100;
                    spanMeter.textContent = 100 + '%';
                    console.log('You have successfully uploaded your file');
                }
            }
        })
        .catch(function(err){
            console.log(err);
            console.log('Error uploading file to the server');
        });
    };

    reader.readAsDataURL( blob );
}

    Errors

    The Car Ad API uses the following error codes:

    Error Code Meaning
    200 Successful Request -- Your request is successful.
    201 Create Successful -- The car ad was successfully created.
    400 Bad Request -- Your request is invalid.
    401 Unauthorized -- Your API key is wrong.
    403 Forbidden -- The car requested is hidden for administrators only.
    404 Not Found -- The specified car could not be found.
    405 Method Not Allowed -- You tried to access a car with an invalid method.
    406 Not Acceptable -- You requested a format that isn't json.
    410 Gone -- The car requested has been removed from our servers.
    418 I'm a teapot.
    429 Too Many Requests -- You're requesting too many cars! Slow down!
    500 Internal Server Error -- We had a problem with our server. Try again later.
    503 Service Unavailable -- We're temporarily offline for maintenance. Please try again later.
